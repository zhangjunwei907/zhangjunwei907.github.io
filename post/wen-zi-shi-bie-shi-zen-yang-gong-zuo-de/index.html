double sample(double y, double x) {
    double sum = 0;
    int yt = (int)y, xt = (int)x;

        int yi = yt + dy, xi = xt + dx;

}

typedef const char *Pos;

int expr(Pos &p);

int term(Pos &p) {
    if(*p == '(') {
        ++ p;
        int x = expr(p);
        ++ p;
        return x;
    }else {
        int x = 0;
        while(isdigit(*p)) {
            x = x * 10 + (*p - '0');
            ++ p;
        }
        return x;
    }
}

int factor(Pos &p) {
    int x = term(p);
    while(*p == '*' || *p == '/') {
        bool div = *p == '/';
        ++ p;
        int y = term(p);
        if(!div) x *= y; else x /= y;
    }
    return x;
}

int expr(Pos &p) {
    int x = factor(p);
    while(*p == '+' || *p == '-') {
        bool sign = *p == '-';
        ++ p;
        int y = factor(p);
        if(!sign) x += y; else x -= y;
    }
    return x;
}
